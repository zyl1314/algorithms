## 题目描述
>给定一个二叉树，检查它是否是它自己的镜像（即，围绕它的中心对称）。
## 示例

```
二叉树 [1,2,2,3,4,4,3] 是对称的。

    1
   / \
  2   2
 / \ / \
3  4 4  3

下面这个 [1,2,2,null,3,null,3] 则不是:

    1
   / \
  2   2
   \   \
   3    3
```
## 代码

```
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isSymmetric = function(root) {
    let arrLeft = [];
    let arrRight = [];
    deep('left')(root, arrLeft);
    deep('right')(root, arrRight);
    
    if (arrLeft.length !== arrRight.length) return false;
    for (let i = 0; i < arrLeft.length; i++) {
        if (arrLeft[i] !== arrRight[i]) return false;
    }
    
    return true;
};

function deep(derection) {
    if (derection == 'left') {
        return function deepFromLeft(root, arr) {
            if (root == null) return;
            arr.push(root.val);
            if (root.left) {
                deepFromLeft(root.left, arr);
            } else {
                arr.push(null);
            }
            if (root.right) {
                deepFromLeft(root.right, arr)
            } else {
                arr.push(null);
            }
        }
    } else {
        return function deepFromRight(root, arr) {
            if (root == null) return;
            arr.push(root.val);
            if (root.right) {
                deepFromRight(root.right, arr);
            } else {
                arr.push(null);
            }
            if (root.left) {
                deepFromRight(root.left, arr)
            } else {
                arr.push(null);
            }
        }        
    }
}
```
