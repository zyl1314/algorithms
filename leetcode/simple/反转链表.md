```
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var reverseList = function(head) {
    if (!head) return null;
    let pre = null;
    while(head) {
        let temp = head.next;
        head.next = pre;
        pre = head;
        if (temp) {
            head = temp;
        } else {
            return head;
        }
    }
};
```
