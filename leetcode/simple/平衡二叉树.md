## 题目描述
>给定一个二叉树，确定它是高度平衡的。
## 示例

```
给出二叉树 [3,9,20,null,null,15,7]:

    3
   / \
  9  20
    /  \
   15   7
返回true

给出二叉树 [1,2,2,3,3,null,null,4,4]:

       1
      / \
     2   2
    / \
   3   3
  / \
 4   4
返回false
```
## 代码

```
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isBalanced = function(root) {
    if (!root) return true;
    let queue = [root];
    while(queue.length) {
        let now = queue.pop();
        if (Math.abs(maxL(now.left) - maxL(now.right)) > 1) return false;
        now.right && queue.push(now.right);
        now.left && queue.push(now.left);
    }
    return true;
};

var maxL = function(root) {
    if (!root) return 0;
    if (maxL(root.left) > maxL(root.right)) {
        return maxL(root.left) + 1;
    } else {
        return maxL(root.right) + 1;
    }
}
```
