## 题目描述
> 给定两个二叉树，写一个函数来检查它们是否相同。

如果两棵树在结构上相同并且节点具有相同的值，则认为它们是相同的。

## 示例

```
输入 :     1          1
          / \       / \
         2   3     2   3

        [1,2,3],   [1,2,3]

输出: true

输入  :    1          1
          /           \
         2             2

        [1,2],     [1,null,2]

输出: false

输入 :     1          1
          / \       / \
         2   1     1   2

        [1,2,1],   [1,1,2]

输出: false
```
## 代码

```
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} p
 * @param {TreeNode} q
 * @return {boolean}
 */
var isSameTree = function(p, q) {
    let pArr = [];
    let qArr = [];
    deep(p, pArr);
    deep(q, qArr);
    if (pArr.length !== qArr.length) return false;
    for (let i = 0; i <= pArr.length; i++) {
        if(pArr[i] !== qArr[i]) return false;
    }
    return true;
};

function deep(p, arr) {
    if (p == null) return;
    arr.push(p.val);
    if (p.left) {
        deep(p.left, arr);
    } else {
        arr.push(null);
    }
    if (p.right) {
        deep(p.right, arr);
    } else {
        arr.push(null);
    }
}
```
