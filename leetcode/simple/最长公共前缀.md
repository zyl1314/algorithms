## 题目描述
> 编写一个函数来查找字符串数组中最长的公共前缀字符串。

## 代码

```
/**
 * @param {string[]} strs
 * @return {string}
 */
var longestCommonPrefix = function(strs) {
    if (!strs.length) return '';
    let result = '';
    for (let i = 0; i < strs[0].length; i++) {
        let flag = false;
        result += strs[0][i];
        flag = strs.every(function(str) {
            return str.indexOf(result) === 0;
        })
        if (!flag) return result.slice(0, -1);
    }
    return result;
};
```
